129                         .Name = name,

133             Dim Nodes = From [property] As PropertyInfo
134                         In type.GetProperties
135                         Let attributes = [property].GetCustomAttributes(attributeType:=ProfileItemNode, inherit:=False)
136                         Where attributes.Length = 1
137                         Select New With {
138                             .[Property] = [property],
139                             .Entity = [property].GetValue(obj, Nothing)
140                         } ' 在这里是用匿名类型而不是直接使用Linq的匿名类型的原因是在后面还需要进行赋值操作，而Linq的匿名类型的属性是ReadOnly的

142             Dim innerNodes = Nodes.ToArray

145                 For Each x In innerNodes

147                     If x.Entity Is Nothing Then
148                         Try

150                         Catch ex As Exception

262             Dim keys As String() = data.Select(Function(x) x.Name).ToArray
263             Dim maxLen As Integer = keys.Select(AddressOf Len).Max
264             Dim sb As New StringBuilder(New String("-"c, 120))

273                     str &= "     // " & line.Description
274                 End If

51     ''' <summary>
52     ''' Windows Metafile (WMF) is an image file format originally designed for Microsoft Windows in the 1990s. 
53     ''' Windows Metafiles are intended to be portable between applications and may contain both vector graphics 
54     ''' and bitmap components. It acts in a similar manner to SVG files.
55     ''' 
56     ''' Essentially, a WMF file stores a list of function calls that have to be issued to the Windows Graphics 
57     ''' Device Interface (GDI) layer to display an image on screen. Since some GDI functions accept pointers 
58     ''' to callback functions for error handling, a WMF file may erroneously include executable code.
59     ''' 
60     ''' WMF Is a 16-bit format introduced in Windows 3.0. It Is the native vector format for Microsoft Office 
61     ''' applications such as Word, PowerPoint, And Publisher. As of 2015 revision 12 of the Windows Metafile 
62     ''' Format specification Is available for online reading Or download as PDF.
63     ''' </summary>
64     ''' <remarks>
65     ''' The original 16 bit WMF file format was fully specified in volume 4 of the 1992 Windows 3.1 SDK documentation
66     ''' (at least if combined with the descriptions of the individual functions and structures in the other volumes), 
67     ''' but that specification was vague about a few details. These manuals were published as printed books available 
68     ''' in bookstores with no click through EULA or other unusual licensing restrictions (just a general warning that 
69     ''' if purchased as part of a software bundle, the software would be subject to one).
70     ''' 
71     ''' Over time the existence Of that historic specification was largely forgotten And some alternative implementations 
72     ''' resorted To reverse engineering To figure out the file format from existing WMF files, which was difficult And 
73     ''' Error prone. In September 2006, Microsoft again published the WMF file format specification In the context Of 
74     ''' the Microsoft Open Specification Promise, promising To Not assert patent rights To file format implementors. 
75     ''' </remarks>

90         ''' <summary>
91         ''' The WMF format was designed to be executed by the Windows GDI layer in order to restore the image, but as 
92         ''' the WMF binary files contain the definition of the GDI graphic primitives that constitute this image, it is 
93         ''' possible to design alternative libraries that render WMF binary files, like the Kyktir application does, or 
94         ''' convert them into other graphic formats. For example, the Batik library is able to render WMF files and 
95         ''' convert them to their Scalable Vector Graphics (SVG) equivalent. The Vector Graphics package of the FreeHEP 
96         ''' Java library allows the saving of Java2D drawings as Enhanced Metafiles (EMF). Inkscape and XnView can export 
97         ''' to WMF or EMF.
98         ''' </summary>
99         ''' <param name="size"></param>
100         ''' <param name="save$"></param>
101         ''' <param name="backgroundColor$"></param>
102         Sub New(size As Size, save$, Optional backgroundColor$ = NameOf(Color.Transparent))
103             Dim bitmap As New Bitmap(size.Width, size.Height)
104
105             gSource = Graphics.FromImage(bitmap)
106             gSource.Clear(backgroundColor.TranslateColor)
107
108             hdc = gSource.GetHdc()
109             size = bitmap.Size
110             curMetafile = New Metafile(save, hdc)
111             Graphics = Graphics.FromImage(curMetafile)
112             Graphics.SmoothingMode = SmoothingMode.HighQuality
113
114             FilePath = save
115         End Sub

263         ''' </param>
264         ''' <param name="files">
265         ''' A set of file names that are to be added
266         ''' </param>
267         ''' <param name="action">
268         ''' Specifies how we are going to handle an existing archive
269         ''' </param>
270         ''' <param name="compression">
271         ''' Specifies what type of compression to use - defaults to Optimal

274         <ExportAPI("Zip.Add.Files", Info:="Allows you to add files to an archive, whether the archive already exists or not")>
275         <Extension>
276         Public Sub AddToArchive(<Parameter("files", "A set of file names that are to be added")> files As IEnumerable(Of String),
277                                 <Parameter("Zip", "The name of the archive to you want to add your files to")> archiveFullName$,
278                                 Optional action As ArchiveAction = ArchiveAction.Replace,
279                                 Optional fileOverwrite As Overwrite = Overwrite.IfNewer,
280                                 Optional compression As CompressionLevel = CompressionLevel.Optimal,
281                                 Optional relativeDIR$ = Nothing)
282
283             'Identifies the mode we will be using - the default is Create
284             Dim mode As ZipArchiveMode = ZipArchiveMode.Create
285
286             'Determines if the zip file even exists
287             Dim archiveExists As Boolean = IO.File.Exists(archiveFullName)
288
289             'Figures out what to do based upon our specified overwrite method
290             Select Case action
291                 Case ArchiveAction.Merge
292                     'Sets the mode to update if the file exists, otherwise
293                     'the default of Create is fine
294                     If archiveExists Then
295                         mode = ZipArchiveMode.Update
296                     End If
297
298                 Case ArchiveAction.Replace
299                     'Deletes the file if it exists.  Either way, the default
300                     'mode of Create is fine
301                     If archiveExists Then
302                         IO.File.Delete(archiveFullName)
303                     End If
304
305                 Case ArchiveAction.[Error]
306                     'Throws an error if the file exists
307                     If archiveExists Then
308                         Throw New IOException($"The zip file {archiveFullName.ToFileURL.CLIPath} already exists.")
309                     End If
310
311                 Case ArchiveAction.Ignore
312                     'Closes the method silently and does nothing
313                     If archiveExists Then
314                         Return
315                     End If
316
317                 Case Else
318
319             End Select
320
321             'Opens the zip file in the mode we specified
322             Using zipFile As ZipArchive = IO.Compression.ZipFile.Open(archiveFullName, mode)
323
324                 'This is a bit of a hack and should be refactored - I am
325                 'doing a similar foreach loop for both modes, but for Create
326                 'I am doing very little work while Update gets a lot of
327                 'code.  This also does not handle any other mode (of
328                 'which there currently wouldn't be one since we don't
329                 'use Read here).
330
331                 If mode = ZipArchiveMode.Create Then
332                     Dim entryName$
333
334                     For Each path As String In files
335                         ' Adds the file to the archive
336                         If relativeDIR.StringEmpty Then
337                             entryName = IO.Path.GetFileName(path)
338                         Else
339                             entryName = RelativePath(relativeDIR, path, appendParent:=False)
340                         End If
341
342                         Call zipFile.CreateEntryFromFile(path, entryName, compression)
343                     Next
344                 Else
345                     For Each path As String In files
346                         Dim fileInZip = (From f In zipFile.Entries Where f.Name = IO.Path.GetFileName(path)).FirstOrDefault()
347
348                         Select Case fileOverwrite
349                             Case Overwrite.Always
350
351                                 'Deletes the file if it is found
352                                 If fileInZip IsNot Nothing Then
353                                     fileInZip.Delete()
354                                 End If
355
356                                 'Adds the file to the archive
357                                 zipFile.CreateEntryFromFile(path, IO.Path.GetFileName(path), compression)
358
359                             Case Overwrite.IfNewer
360
361                                 'This is a bit trickier - we only delete the file if it is
362                                 'newer, but if it is newer or if the file isn't already in
363                                 'the zip file, we will write it to the zip file
364
365                                 If fileInZip IsNot Nothing Then
366
367                                     'Deletes the file only if it is older than our file.
368                                     'Note that the file will be ignored if the existing file
369                                     'in the archive is newer.
370                                     If fileInZip.LastWriteTime < IO.File.GetLastWriteTime(path) Then
371                                         fileInZip.Delete()
372
373                                         'Adds the file to the archive
374                                         zipFile.CreateEntryFromFile(path, IO.Path.GetFileName(path), compression)
375                                     End If
376                                 Else
377                                     'The file wasn't already in the zip file so add it to the archive
378                                     zipFile.CreateEntryFromFile(path, IO.Path.GetFileName(path), compression)
379                                 End If
380
381                             Case Overwrite.Never
382
383                                 'Don't do anything - this is a decision that you need to
384                                 'consider, however, since this will mean that no file will
385                                 'be writte.  You could write a second copy to the zip with
386                                 'the same name (not sure that is wise, however).
387
388                             Case Else
389
390                         End Select
391                     Next

471         ''' <summary>
472         ''' Trim the CLI argument name its prefix symbols.
473         ''' (修剪命令行参数名称的前置符号)
474         ''' </summary>
475         ''' <param name="argName"></param>
476         ''' <returns></returns>
477         <ExportAPI("Trim.Prefix.BoolFlag")>
478         <Extension>
479         Public Function TrimParamPrefix(argName$) As String
480             If argName.StartsWith("--") Then
481                 Return Mid(argName, 3)
482             ElseIf argName.StartsWith("-") OrElse argName.StartsWith("\") OrElse argName.StartsWith("/") Then
483                 Return Mid(argName, 2)
484             Else
485                 Return argName
486             End If
487         End Function

489         ''' <summary>
490         ''' 请注意，这个是有方向性的，由于是依照参数1来进行比较的，假若args2里面的参数要多于第一个参数，但是第一个参数里面的所有参数值都可以被参数2完全比对得上的话，就认为二者是相等的
491         ''' </summary>
492         ''' <param name="args1"></param>
493         ''' <param name="args2"></param>
494         ''' <returns></returns>
495         ''' 
496         <ExportAPI("CLI.Equals")>
497         Public Function Equals(args1 As CommandLine, args2 As CommandLine) As Boolean
498             If Not String.Equals(args1.Name, args2.Name, StringComparison.OrdinalIgnoreCase) Then
499                 Return False
500             End If
501
502             For Each bFlag As String In args1.BoolFlags
503                 If Not args2.GetBoolean(bFlag) Then
504                     Return False
505                 End If
506             Next
507
508             For Each arg As NamedValue(Of String) In args1.__arguments
509                 Dim value2 As String = args2(arg.Name)
510                 If Not String.Equals(value2, arg.Value, StringComparison.OrdinalIgnoreCase) Then
511                     Return False
512                 End If
513             Next
514
515             Return True
516         End Function
